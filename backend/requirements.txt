# requirements.txt — Python dependencies for PaperChat backend
#
# Install everything with:
#   pip install -r requirements.txt
#
# A note on versions:
# We pin major versions (fastapi>=0.110) but allow minor updates.
# This gives you security patches without breaking changes.

# ── Web Framework ────────────────────────────────────────────────────────────
fastapi>=0.110.0          # The API framework — like Express.js but Python
uvicorn[standard]>=0.29.0 # ASGI server that runs FastAPI (like Node for Express)
python-multipart>=0.0.9   # Required for FastAPI to handle file uploads (FormData)

# ── PDF Processing ────────────────────────────────────────────────────────────
pdfplumber>=0.10.3        # Extracts text from PDFs. Handles columns, tables, etc.
                          # Much better than PyPDF2 for real-world messy PDFs.

# ── Text Splitting ────────────────────────────────────────────────────────────
langchain-text-splitters>=0.0.1  # Just the chunking utility from LangChain.
                                  # We don't need all of LangChain — just this one piece.
                                  # It handles overlapping chunks cleanly.

# ── Embeddings ────────────────────────────────────────────────────────────────
sentence-transformers>=2.7.0  # Runs embedding models locally. Free, no API key needed.
                               # We use all-MiniLM-L6-v2: fast, small (80MB), great quality.
torch>=2.0.0                  # PyTorch — the deep learning library sentence-transformers runs on.
                               # CPU-only is fine for embeddings, no GPU needed.

# ── Vector Database ───────────────────────────────────────────────────────────
chromadb>=0.4.24          # Local vector database. Stores embeddings and lets us
                          # do similarity search. Runs entirely in-process, no
                          # server needed. Data persists to disk automatically.

# ── LLM (Groq) ────────────────────────────────────────────────────────────────
groq>=0.7.0               # Official Groq Python SDK. Supports streaming out of the box.
                          # Groq runs Llama 3.3 70B at ~800 tokens/sec — insanely fast.

# ── Utilities ─────────────────────────────────────────────────────────────────
python-dotenv>=1.0.0      # Loads .env files into environment variables.
                          # Keeps API keys out of your code.
pydantic>=2.0.0           # Data validation library. FastAPI is built on top of it.
                          # Automatically validates request/response shapes.
